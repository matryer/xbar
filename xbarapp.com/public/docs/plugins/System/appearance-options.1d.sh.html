# APPLESCRIPT FUNCTIONS
function changeDesktopImg () {
osascript <<EOD
	tell application "Finder"
		set img to some file of folder (POSIX file "$1") as string
	end tell
	tell application "System Events"
		tell current desktop
			set pictures folder to "$1"
			set picture to img
		end tell
	end tell
EOD
}
function toggleDarkMode () {
osascript <<EOD
	tell application "System Events" to tell appearance preferences
		set dark mode to not dark mode
	end tell
EOD
}
function restartBitBar () {
osascript <<EOD
	tell application "BitBar" to quit
	delay 1
	tell application "BitBar" to activate
EOD
}
function launchScreensaver () {
osascript <<EOD
	try
		tell application id "com.apple.ScreenSaver.Engine" to launch
	end try
EOD
}
function desktopImgRotation () {
# $1 = picture rotation mode (0=off, 1=interval, 2=login, 3=sleep)
# $2 = picture rotation timing (in seconds)
osascript <<EOD
	tell application "System Events"
		tell current desktop
			set picture rotation to "$1"
			set change interval to "$2"
		end tell
	end tell
EOD
}
function checkRotationTiming () {
osascript <<EOD
	tell application "System Events"
		tell current desktop
			get properties
			return change interval
		end tell
	end tell
EOD
}
function checkRandom () {
osascript <<EOD
	tell application "System Events"
		tell current desktop
			get properties
			return random order
		end tell
	end tell
EOD
}
function toggleRandomOrder () {
osascript <<EOD
	tell application "System Events"
		tell current desktop
			get properties
			set random order to not random order
		end tell
	end tell
EOD
}
function nightShift () {
osascript <<EOD
	set nshiftOptions to { 0, 25, 50, 75, 100 }
	set nshiftDialog to (choose from list nshiftOptions with prompt "Choose warmth level:" default items "0" with title "Night Shift Settings")
	if nshiftDialog is not false then
		do shell script "$nshiftDir" & " " & nshiftDialog
		return
	end if
EOD
}
